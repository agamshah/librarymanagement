(

member_type varchar(20),
member_name varchar(20),
no_of_copies number(5),
primary key(member_code),
CONSTRAINT member_type_check CHECK(member_type IN ('student', 'faculty'))

create table book_issue
(
book_issue number(5) primary key NOT NULL,
date_of_issue date,
date_of_return date,
book_code number(5),
member_code number(5),
constraint book_code_fk FOREIGN KEY(book_code) REFERENCES book(book_code),
constraint member_code_fk FOREIGN KEY(member_code) REFERENCES member(member_code)
);
